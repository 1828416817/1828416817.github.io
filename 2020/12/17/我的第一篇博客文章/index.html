我的第一篇博客文章

package main

import (
“fmt”
“sync”
)
//定义全局同步互斥锁变量解决并发资源竞争问题
var lock sync.Mutex
var(
myMap = make(map[int]float64)
wg sync.WaitGroup
)
func Factorial(n int) {
res:=1
for i:=1;i<=n ;i++ {
res*=i
}
//对要对map进行写入的操作进行加锁,防止出现并发状态下资源竞争导致出错
lock.Lock()
myMap[n]=float64(res)
//解锁.让队列中排队的线程可以加入
lock.Unlock()
wg.Done()

}
func main() {
for i:=1;i<=20 ;i++ {
wg.Add(1)
go Factorial(i)
}
//等待所有协程执行完毕后再结束主线程
wg.Wait()
lock.Lock()
for i, i2 := range myMap {
fmt.Println(i,i2)
}
lock.Unlock()
}
